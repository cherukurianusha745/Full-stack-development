Write a program which prints small/large/equal relation of given two integers a and b.

Input
Two integers a and b separated by a single space are given in a line.

Output
For given two integers a and b, print

a < b
if a is less than b,

a > b
if a is greater than b, and

a == b
if a equals to b.

Constraints
-1000 ≤ a, b ≤ 1000
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        String operations = (a < b) ? "a < b" :
                        (a > b) ? "a > b" :
                                  "a == b";

        System.out.println(operations);
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int T = sc.nextInt(); 

        for (int i = 0; i < T; i++) {
            int A = sc.nextInt(); 
            int B = sc.nextInt(); 
            int C = sc.nextInt(); 
            int D = sc.nextInt(); 

            int costFirst = A - C;
            int costSecond = B - D;

            if (costFirst < costSecond) {
                System.out.println("First");
            } else if (costFirst > costSecond) {
                System.out.println("Second");
            } else {
                System.out.println("Any");
            }
        }
    }
}
